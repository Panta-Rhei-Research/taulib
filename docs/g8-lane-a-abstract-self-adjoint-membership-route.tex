\section*{Line-by-Line Proof}

\begin{enumerate}
\item Fix an actual nonzero-height \(\xi\)-carrier \(z\), with orthodox point
  \(s_z=\sigma_z+i\gamma_z\).

\item By the canonical scaled image law,
  \[
    V(z)=c_\tau\left(s_z(1-s_z)-\frac14\right).
  \]

\item By canonical abstract spectral membership,
  \[
    \Mem_{H_L}(V(z)).
  \]

\item By self-adjoint spectral reality for genuine spectral members,
  \[
    \im V(z)=0.
  \]

\item Since \(c_\tau\in\mathbb{R}\) and \(c_\tau\ne0\), scale cancellation gives
  \[
    \im\left(s_z(1-s_z)-\frac14\right)=0.
  \]
  This is exactly the Lean adapter already isolated by the scaled
  operator-image modules.

\item Expand the centered quadratic:
  \[
    s_z(1-s_z)-\frac14
    =
    \left(\sigma_z(1-\sigma_z)+\gamma_z^2-\frac14\right)
    +i\gamma_z(1-2\sigma_z).
  \]
  Hence
  \[
    \im\left(s_z(1-s_z)-\frac14\right)
    =
    \gamma_z(1-2\sigma_z).
  \]

\item Therefore
  \[
    \gamma_z(1-2\sigma_z)=0.
  \]

\item Since \(z\) is a nonzero-height carrier, \(\gamma_z\ne0\).

\item Hence \(1-2\sigma_z=0\), so
  \[
    \sigma_z=\frac12.
  \]

\item Because \(z\) was arbitrary, the nonzero-height Lane A spectral reality
  target follows.
\end{enumerate}
